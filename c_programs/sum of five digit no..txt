#include <stdio.h>
#include <string.h>
#include <math.h>
#include <stdlib.h>

int main() {
	
    int n;
    scanf("%d", &n);
    int dig, temp,sum=0;
    temp=n;
    //code to calculate the sum of the five digits on n.
    while(temp>0)
    {
        dig=temp%10;
        sum=sum+dig;
        temp=temp/10;
    }
    printf("%d\n",sum);

    

    return 0;
